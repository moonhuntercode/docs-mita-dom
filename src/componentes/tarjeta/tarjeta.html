import { Signal } from 'mita-dom';
import { estadoAppGlobal } from '../../store/appStore.js';

export class MitaTarjeta extends MitaElement {
  constructor() {
    super();
    this.estadoLocal = new Signal(0);
  }

  iniciarLogica() {
    estadoAppGlobal.suscribir(estado => {
      this.$valorGlobal.textContent = estado.visitas;
    });

    this.estadoLocal.suscribir(valor => {
      this.$valorLocal.textContent = valor;
    });
  }
}
