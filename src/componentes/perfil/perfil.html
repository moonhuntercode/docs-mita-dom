import { crearRecurso } from 'mita-dom';

export class MitaPerfil extends MitaElement {
  constructor() {
    super();
    this.recursoPerfil = crearRecurso(async () => {
      const res = await fetch('https://jsonplaceholder.typicode.com/users/1');
      return await res.json();
    });
  }

  iniciarLogica() {
    this.recursoPerfil.suscribir(estado => {
      if (estado.loading) this.mostrarLoader();
      else if (estado.error) this.mostrarError();
      else this.renderData(estado.data);
    });
  }
}
