import { rutaActual } from 'mita-dom';

export class MitaPerfil extends MitaElement {
  iniciarLogica() {
    this.$tabInfo = this.querySelector('#tab-info');
    this.$tabSeguridad = this.querySelector('#tab-seguridad');

    // Sub-Router Local (Nested Routes)
    rutaActual.suscribir(ruta => {
      if (!ruta.startsWith('/perfil')) return;
      
      // Esto emula el  de React Router
      this.$tabInfo.style.display = ruta === '/perfil/info' ? 'block' : 'none';
      this.$tabSeguridad.style.display = ruta === '/perfil/seguridad' ? 'block' : 'none';
    });
  }
}

Buenas Prácticas Frontend: Para hacer Nested Routes no necesitas un framework inmenso. Solo suscríbete al Signal del router global dentro del componente y decide qué sub-bloque de HTML mostrar (tu propio Outlet). Esto respeta el encapsulamiento.
