import { estadoAppGlobal } from '../../store/appStore.js';

export class MitaFormulario extends MitaElement {
  iniciarLogica() {
    this.$form.addEventListener('submit', (e) => {
      e.preventDefault();
      const nombre = this.$input.value;

      // Mutación Granular y Segura del Estado Global
      estadoAppGlobal.patch({ 
        usuario: { nombre: nombre } 
      });
      
      this.mostrarFeedback('¡Estado Global Actualizado!');
    });
  }
}
