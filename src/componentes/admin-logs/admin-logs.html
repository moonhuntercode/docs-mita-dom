// router.js
import { rutaActual } from 'mita-dom';

const registroRutas = [
  { path: '/admin/logs', tag: 'demo-admin-logs', meta: { requiresAuth: true } }
];

// 🛡️ Navigation Guard Interceptor
rutaActual.suscribir(ruta => {
  const config = registroRutas.find(r => r.path === ruta);
  
  if (config?.meta?.requiresAuth && !localStorage.getItem('mita_token')) {
    alert("Acceso Denegado. Interceptado por MitaDOM Guard.");
    navegarA('/perfil');
    return; // ⛔ Bloquea carga del Chunk
  }
});

Buenas Prácticas Frontend: Para llegar a esta vista, el Router de MitaDOM evaluó que tuvieras un token local. Este es el patrón estándar de Navigation Guards que encuentras en Vue Router, implementado con 0 dependencias.
