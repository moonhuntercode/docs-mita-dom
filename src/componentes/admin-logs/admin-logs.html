import { StorageAdapter } from 'mita-dom';

export class DemoAdminLogs extends MitaElement {
  iniciarLogica() {
    const db = new StorageAdapter('mita_logs');
    
    this.$btnLimpiar.addEventListener('click', async () => {
      await db.clear();
      this.cargarLogs();
    });

    this.cargarLogs();
  }
}
