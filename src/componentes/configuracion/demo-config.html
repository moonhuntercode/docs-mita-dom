import { estadoConfiguracion } from '../../store/docsStore.js';

export class DemoConfig extends MitaElement {
  iniciarLogica() {
    const pos = estadoConfiguracion.get().sidebarPos;
    
    this.$radios.forEach(radio => {
      radio.addEventListener('change', (e) => {
        estadoConfiguracion.patch({ sidebarPos: e.target.value });
      });
    });
  }
}
